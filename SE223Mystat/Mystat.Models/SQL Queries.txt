CREATE DATABASE SE223

USE SE223

CREATE TABLE Student
(
    Id INT PRIMARY KEY IDENTITY(1,1),
    FirstName NVARCHAR(50) NOT NULL,
    LastName NVARCHAR(50) NOT NULL,
    RegisterDate DATETIME NOT NULL DEFAULT(GETDATE()),
    FullName AS FirstName + LastName,
    Attends BIT NOT NULL,
    AttendsOnline BIT NOT NULL,
    Brilliants INT NULL,
    Comment NVARCHAR(MAX) NULL
)

CREATE PROCEDURE sp_addNewStudent
@param0 NVARCHAR(50),
@param1 NVARCHAR(50),
@param2 BIT,
@param3 BIT,
@param4 INT,
@param5 NVARCHAR(MAX)
AS
BEGIN
    INSERT INTO Student(FirstName,LastName,Attends,AttendsOnline,Brilliants,Comment)
    VALUES(@param0,@param1,@param2,@param3,@param4,@param5)
END



CREATE PROCEDURE sp_getAllStudents
AS
BEGIN
    SELECT
    Id,
    FirstName,
    LastName,
    RegisterDate,
    FullName,
    Attends,
    AttendsOnline,
    Brilliants,
    Comment
FROM Student
END



CREATE PROCEDURE sp_getStudentById
@param0 INT
AS
BEGIN
    SELECT
        Id,
        FirstName,
        LastName,
        RegisterDate,
        FullName,
        Attends,
        AttendsOnline,
        Brilliants,
        Comment
    FROM Student
    WHERE Id = @param0
END





CREATE PROCEDURE sp_updateStudent
@param0 INT,
@param1 NVARCHAR(50),
@param2 NVARCHAR(50),
@param3 DATETIME,
@param4 BIT,
@param5 BIT,
@param6 INT,
@param7 NVARCHAR(MAX)
AS
BEGIN
    UPDATE Student
    SET
        FirstName = @param1,
        LastName = @param2,
        RegisterDate = @param3,
        Attends = @param4,
        AttendsOnline = @param5,
        Brilliants = @param6,
        Comment = @param7
    WHERE Id = @param0
END

CREATE PROCEDURE sp_deleteStudent
@param0 INT
AS
BEGIN
    DELETE FROM Student
    WHERE Id = @param0
END






CREATE TABLE Teachers
(
	Id INT PRIMARY KEY IDENTITY NOT NULL,
	FirstName NVARCHAR(50) NOT NULL,
	LastName NVARCHAR(50) NOT NULL,
	Pin CHAR(11) NOT NULL,
	PhoneNumber CHAR(9) NOT NULL,
	Email VARCHAR(MAX) NOT NULL
)

CREATE TABLE Subjects(
	[Id] INT PRIMARY KEY IDENTITY(1,1) NOT NULL,
	[Title] [nvarchar](100) NOT NULL
)







CREATE PROCEDURE sp_getAllSubjects
AS
BEGIN
	SELECT
		Id,
		Title
	FROM Subjects
END


CREATE PROCEDURE sp_getSubjectById
@param0 INT
AS
BEGIN
	SELECT
		Id,
		Title
	FROM Subjects
	WHERE Id = @param0
END


CREATE PROCEDURE sp_addNewSubject
@param0 NVARCHAR(100)
AS
BEGIN
	INSERT INTO Subjects(Title)
	VALUES(@param0)
END


CREATE PROCEDURE sp_deleteSubject
@param0 INT
AS
BEGIN
	DELETE
	FROM Subjects
	WHERE Id = @param0
END

CREATE PROCEDURE sp_updateSubject
@param0 NVARCHAR(100),
@param1 INT
AS
BEGIN
	UPDATE Subjects
	SET Title = @param0
	WHERE Id = @param1
END








CREATE TABLE Groups
(
    Name VARCHAR(100) NOT NULL,
	SubjectId INT FOREIGN KEY REFERENCES Subjects(Id),
	TeacherId INT FOREIGN KEY REFERENCES Teachers(Id),
	StudentId INT FOREIGN KEY REFERENCES Student(Id),
)


CREATE PROCEDURE sp_addGroup
@param0 VARCHAR(100),
@param1 INT,
@param2 INT,
@param3 INT
AS
BEGIN
	INSERT INTO Groups(Name,SubjectId,TeacherId,StudentId)
	VALUES(@param0,@param1,@param2,@param3)
END



CREATE PROCEDURE [dbo].[sp_updateSubjectForGroup]
@param0 VARCHAR(100),
@param1 INT,
@param2 INT
AS
BEGIN
	UPDATE Groups
	SET SubjectId =@param2
	WHERE SubjectId = @param1 AND Name = @param0
END



CREATE PROCEDURE [dbo].[sp_updateTeacherForGroup]
@param0 VARCHAR(100),
@param1 INT,
@param2 INT
AS
BEGIN
	UPDATE Groups
	SET TeacherId =@param1
	WHERE TeacherId = @param2  AND [Name] = @param0
END



CREATE PROCEDURE [dbo].[sp_updateStudentForGroup]
@param0 INT,
@param1 INT,
@param2 VARCHAR(100)
AS
BEGIN
    UPDATE Groups
    SET StudentId = @param1
    WHERE StudentId = @param0 AND [Name]=@param2
END



CREATE PROCEDURE [dbo].[sp_deleteStudentForGroup]
@param0 INT,
@param1 VARCHAR(100)
AS
BEGIN
    DELETE FROM Groups
    WHERE StudentId = @param0 AND [Name] = @param1
END



CREATE PROCEDURE sp_deleteGroup
@param0 VARCHAR(100)
AS
BEGIN
    DELETE FROM Groups
    WHERE Name = @param0
END
